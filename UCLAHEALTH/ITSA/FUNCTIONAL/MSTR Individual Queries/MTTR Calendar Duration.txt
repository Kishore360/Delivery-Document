
select duration_c/Resolved_Incidents from 
( Select  count(a11.row_key) AS Resolved_Incidents,
 sum((a11.duration_c) / 86400.0) AS duration_c
from ldb.f_incident a11
join ldb.d_calendar_date a111
   on  (a11.opened_on_key = a111.row_key)
join ldb.d_calendar_month a112
   on  (a111.month_start_date_key = a112.row_key)
where a112.lagging_count_of_month between 0 and 11)
