 select'f_request_task' as Table_Name, count(a11.row_key) Row_Count
 from  (  select row_key FROM intuit_mdwdb.f_request_task rt  WHERE rt.soft_deleted_flag = 'N')             a11 
 union
 select'd_calendar_date' as Table_Name, count(a11.row_key) Row_Count
 from  (  select opened_on_key,row_key FROM intuit_mdwdb.f_request_task rt  WHERE rt.soft_deleted_flag = 'N')               a11 
join   (SELECT d_calendar_date.row_key,d_calendar_date.month_start_date_key
   FROM bhn_mdwdb.d_calendar_date d_calendar_date
   
) a12
on (a11.opened_on_key = a12.row_key)
union
select'd_calendar_month' as Table_Name, count(a11.row_key) Row_Count
 from  (  select opened_on_key,row_key FROM intuit_mdwdb.f_request_task rt  WHERE rt.soft_deleted_flag = 'N')               a11 
join   (SELECT d_calendar_date.row_key,d_calendar_date.month_start_date_key
   FROM bhn_mdwdb.d_calendar_date d_calendar_date
   
) a12
on (a11.opened_on_key = a12.row_key)
join  (SELECT d_calendar_date.row_key
   FROM bhn_mdwdb.d_calendar_date d_calendar_date
    where d_calendar_date.source_id = 0
 )       a124
on (a12.month_start_date_key = a124.row_key)
